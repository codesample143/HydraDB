cmake_minimum_required(VERSION 3.15)
project(HydraDB)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include directories
include_directories(
    ${CMAKE_SOURCE_DIR}
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/database
    ${CMAKE_SOURCE_DIR}/lib/sql-parser-main/src
    ${CMAKE_SOURCE_DIR}/lib/sql-parser-main/src/parser
    ${CMAKE_SOURCE_DIR}/lib/sql-parser-main/src/sql
)

# Collect source files
file(GLOB PARSER_SRC
    lib/sql-parser-main/src/parser/*.cpp
    lib/sql-parser-main/src/sql/*.cpp
)

set(SRC_FILES
    src/main.cpp
    database/InputBuffer.cpp
    lib/sql-parser-main/src/SQLParser.cpp
    lib/sql-parser-main/src/SQLParserResult.cpp
    lib/sql-parser-main/src/sql/CreateStatement.cpp
    ${PARSER_SRC}
)

add_executable(HydraDB ${SRC_FILES})
